    Common Error Patterns with Solutions

    Common Rust Lifetime Errors: Causes, Example Compiler Messages, and Debugging Techniques

    Recognizing and solving the most frequent lifetime error patterns:

    fn demonstrate_common_error_patterns() {
    println!("🎯 Common Error Patterns - Professional Troubleshooting Guide");
    println!("{:=<70}", "");

    // Common patterns are like recurring operational issues in restaurants
    println!("📋 Most Frequent Lifetime Error Patterns:");

    // Pattern 1: Returning References to Local Variables
    println!("\n1️⃣ Pattern: Returning References to Local Variables");

    println!("   🚫 THE PROBLEM:");
    println!("
   // ❌ This pattern ALWAYS fails:
   fn create_local_reference() -> &String {{
       let local_data = String::from(\"Local menu item\");
       &local_data  // ERROR: local_data dropped at function end
   }}

   ERROR MESSAGE:
   error[E0515]: cannot return reference to local variable `local_data`
   help: consider returning the owned value");

    println!("   ✅ SOLUTIONS:");

    // Solution A: Return owned data
    fn return_owned_data() -> String {
        let local_data = String::from("Owned menu item");
        local_data  // Transfer ownership
    }

    // Solution B: Use static data
    fn return_static_reference() -> &'static str {
        "Static menu item"  // Lives for entire program
    }

    // Solution C: Accept and return reference with same lifetime
    fn process_external_reference(input: &str) -> &str {
        // Process the input and return part of it
        &input[0..input.len().min(10)]
    }

    let owned = return_owned_data();
    let static_ref = return_static_reference();
    let processed = process_external_reference("Long menu description");

    println!("     Solution A (Owned): {}", owned);
    println!("     Solution B (Static): {}", static_ref);
    println!("     Solution C (Processed): {}", processed);

    // Pattern 2: Borrow Conflicts
    println!("\n2️⃣ Pattern: Borrow Conflicts (Mutable + Immutable)");

    println!("   🚫 THE PROBLEM:");
    println!("
   // ❌ This pattern creates conflicts:
   let mut restaurant_data = RestaurantData::new();
   let menu_ref = &restaurant_data.menu;           // Immutable borrow
   restaurant_data.add_item(new_item);             // ERROR: Mutable borrow
   println!(\"Menu: {{:?}}\", menu_ref);             // Immutable borrow used here

   ERROR MESSAGE:
   error[E0502]: cannot borrow `restaurant_data` as mutable because it is also borrowed as immutable");

    struct RestaurantData {
        menu: Vec<String>,
        revenue: f64,
    }

    impl RestaurantData {
        fn new() -> Self {
            RestaurantData {
                menu: Vec::new(),
                revenue: 0.0,
            }
        }

        fn add_item(&mut self, item: String) {
            self.menu.push(item);
        }

        fn get_menu(&self) -> &Vec<String> {
            &self.menu
        }
    }

    println!("   ✅ SOLUTIONS:");

    // Solution A: Limit borrow scope
    let mut restaurant_data = RestaurantData::new();
    {
        let menu_ref = restaurant_data.get_menu();
        println!("     Current menu size: {}", menu_ref.len());
    } // menu_ref goes out of scope here
    restaurant_data.add_item("New Pizza".to_string());
    println!("     Solution A: Limited borrow scope - {} items", restaurant_data.menu.len());

    // Solution B: Clone when needed
    let mut restaurant_data2 = RestaurantData::new();
    let menu_snapshot = restaurant_data2.get_menu().clone();
    restaurant_data2.add_item("Another Pizza".to_string());
    println!("     Solution B: Cloned data - snapshot: {:?}, current: {}",
             menu_snapshot, restaurant_data2.menu.len());

    // Pattern 3: Missing Lifetime Parameters
    println!("\n3️⃣ Pattern: Missing Lifetime Parameters");

    println!("   🚫 THE PROBLEM:");
    println!("
   // ❌ Compiler can't infer relationships:
   fn choose_menu_item(item1: &str, item2: &str) -> &str {{
       if item1.len() > item2.len() {{ item1 }} else {{ item2 }}
   }}

   ERROR MESSAGE:
   error[E0106]: missing lifetime specifier
   help: this function's return type contains a borrowed value,
         but the signature does not say whether it is borrowed from `item1` or `item2`");

    println!("   ✅ SOLUTIONS:");

    // Solution A: Single lifetime parameter (most common)
    fn choose_menu_item_single<'a>(item1: &'a str, item2: &'a str) -> &'a str {
        if item1.len() > item2.len() { item1 } else { item2 }
    }

    // Solution B: Multiple lifetime parameters (when needed)
    fn choose_by_first<'a, 'b>(first: &'a str, _second: &'b str) -> &'a str {
        first  // Always return the first one
    }

    // Solution C: Return owned data to avoid lifetimes
    fn choose_and_format(item1: &str, item2: &str) -> String {
        let chosen = if item1.len() > item2.len() { item1 } else { item2 };
        format!("Selected: {}", chosen)
    }

    let item1 = "Short";
    let item2 = "Much longer item";

    println!("     Solution A (Single lifetime): {}",
             choose_menu_item_single(item1, item2));
    println!("     Solution B (Multiple lifetimes): {}",
             choose_by_first(item1, item2));
    println!("     Solution C (Owned result): {}",
             choose_and_format(item1, item2));

    // Pattern 4: Struct Lifetime Issues
    println!("\n4️⃣ Pattern: Struct Lifetime Issues");

    println!("   🚫 THE PROBLEM:");
    println!("
   // ❌ Struct holds references without lifetime parameters:
   struct Restaurant {{
       name: &str,        // ERROR: missing lifetime parameter
       location: &str,    // ERROR: missing lifetime parameter
   }}

   ERROR MESSAGE:
   error[E0106]: missing lifetime specifier");

    println!("   ✅ SOLUTIONS:");

    // Solution A: Add lifetime parameters
    #[derive(Debug)]
    struct RestaurantRef<'a> {
        name: &'a str,
        location: &'a str,
    }

    impl<'a> RestaurantRef<'a> {
        fn new(name: &'a str, location: &'a str) -> Self {
            RestaurantRef { name, location }
        }

        fn describe(&self) -> String {
            format!("{} is located in {}", self.name, self.location)
